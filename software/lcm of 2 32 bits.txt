 AREA gcdcal,CODE,READONLY
 ENTRY
 MOV R0,#0x12;
 MOV R1,#0x0F;
 MUL R3,R0,R1;
GCD
WHILE CMP R0,R1
	BEQ ENDW
	BGT COND1
	B COND2
COND1 SUB R0,R1
	B GCD
COND2 SUB R1,R0
	B GCD
	B WHILE
ENDW
	MOV R4,R0;
	MOV R5,0x00;
	MOVR6,R3;
DIVIDE CMP R6,R4
	VLT STOP
	ADD R5,R5,#1
	SUB R6,R6,R4
	BAL DIVIDE
STOP B STOP
	END
