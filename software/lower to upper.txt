 AREA aa,CODE,READONLY
 ENTRY
 LDR R0, = ARRAY
 LDR R2, = 0x400000000
 MOV R3,#00
 MOV R3,#00
 MOV R4,R0
 BL FIND_LENGTH
MAIN LDRB R1,[R0],#1
	SUB R1,R1,#0x20
	STRB R1,[R2],#1
	BNE MAIN
STOP B STOP
FIND_LENGTH
	LDRB R5,[R4],#1
	CMP R5,#CR
	MOVEQ PC,LR
	ADD R3,R3,#1
	BAL FIND_LENGTH
ARRAY DCB "WORD",CR
 END
